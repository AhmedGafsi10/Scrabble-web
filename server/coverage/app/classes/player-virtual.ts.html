const distribution = <T>(array: T[], weights: number[], size: number): T[] => {
    const elements:T[] = [];
    const sum = weights.reduce((a, b) => a + b);
    for (let i = 0; i < array.length; ++i) {
        const count = (weights[i] / sum) * size;
        for (let j = 0; j < count; ++j) {
            elements.push(array[i]);
        }
    }
    return elements;
};
 
const randomValue = (distributionData: unknown[]) => {
    const index = Math.floor(distributionData.length * Math.random()); // random index
    return distributionData[index];
};

    playAction(): string {
        const centerCaseCol = 8;
        const centerNode = this.board.getNode('h', centerCaseCol);
        if (centerNode.content) this.findBestPlacements();
        else this.findFirstPlacement();
        return this.playBestAction() as string;
        // this.updateRack();
    }
 
    private findBestPlacements() {
        this.possiblePlacements = new Map();
        for (let i = 0; i < this.board.translator.caseCount; i++) {
            // This is very important: VP should not directly access the cases of the board.
            const caseContent = this.board.getCaseContent(i);
            if (caseContent && caseContent.content) {
                const hNodeStream = this.detectNodeStream(caseContent, PlacementDirections.Horizontal);
                const vNodeStream = this.detectNodeStream(caseContent, PlacementDirections.Vertical);
 
                if (hNodeStream) this.registerHorizontalPlacements(hNodeStream);
                if (vNodeStream) this.registerVerticalPlacements(vNodeStream);
            }
        }
    }
 
    private findFirstPlacement() {
        this.possiblePlacements = new Map();
        const probability = randomValue(distribution([50, 100], [50, 50], 10));
        const placement = {
            row: 'h',
            col: 8,
            direction: 'v',
            oldWord: '',
            newWord: '',
            letters: '',
        } as UserPlacement;
        if (probability === 50) placement.direction = PlacementDirections.Vertical;
        const words = this.board.getDictionaryReader().getAllWords();
 
        const rackLetters = this.rack.getLetters();
        for (const letter of rackLetters) {
            let partial = letter;
            for (const suffix of rackLetters) {
                partial += suffix;
                for (const word of words) {
                    if (word === partial) break;
                }
            }
        }
        const possibleWords = this.findPossibleWords(placement.newWord, this.copyRack().getLetters());
        for (const word of possibleWords) {
            if (word.length < 3) continue;
            placement.newWord = word;
            placement.letters = word;
            this.registerPossiblePlacements(placement);
        }
    }
 
    private detectNodeStream(node: BoardNode, direction: PlacementDirections): NodeStream | undefined {
        let startNode = node;
        let previousNode;
        if (direction === PlacementDirections.Horizontal) {
            previousNode = node.getNeighbor(Directions.Left);
            while (previousNode) {
                startNode = previousNode;
                previousNode = previousNode.getNeighbor(Directions.Left);
            }
        } else {
            previousNode = node.getNeighbor(Directions.Up);
            while (previousNode) {
                startNode = previousNode;
                previousNode = previousNode.getNeighbor(Directions.Up);
            }
        }
        return new NodeStream(startNode);
    }
 
    private registerHorizontalPlacements(hNodeStream: NodeStream) {
        if (!hNodeStream) return;
        hNodeStream.elaborateNodesFlow(PlacementDirections.Horizontal);
        const startNode = hNodeStream.getStartNode();
        const horizontalWord = hNodeStream.getWord(PlacementDirections.Horizontal) as string;
        if (horizontalWord.length > 1) {
            const similarWordsH = this.findSimilarWords(horizontalWord).filter((element) => element !== horizontalWord);
            for (const word of similarWordsH) {
                const wordIndex = word.search(horizontalWord);
                const colIndex = this.board.translator.findColumnIndex(startNode.key) as number;
                const finalColIndex = wordIndex > 0 ? colIndex - wordIndex : colIndex + horizontalWord.length;
                // const finalColIndex = wordIndex > 0 ? colIndex - wordIndex : colIndex;
                const placement = {
                    row: this.board.translator.findRowLetter(startNode.key),
                    col: finalColIndex,
                    direction: PlacementDirections.Horizontal,
                    newWord: word,
                    oldWord: horizontalWord,
                    letters: word.replace(horizontalWord, ''),
                } as UserPlacement;
                if (finalColIndex > 0 && finalColIndex + placement.letters.length <= this.board.translator.columnCount)
                    this.registerPossiblePlacements(placement);
            }
        }
    }
 
    private registerVerticalPlacements(vNodeStream: NodeStream) {
        if (!vNodeStream) return;
        vNodeStream.elaborateNodesFlow(PlacementDirections.Vertical);
        const startNode = vNodeStream.getStartNode();
        const verticalWord = vNodeStream.getWord(PlacementDirections.Vertical) as string;
        if (verticalWord.length > 1) {
            const similarWordsV = this.findSimilarWords(verticalWord).filter((element) => element !== verticalWord);
            for (const word of similarWordsV) {
                const wordIndex = word.search(verticalWord);
                const rowIndex = IndexIterator.getRowIndex(this.board.translator.findRowLetter(startNode.key) as string);
                const finalRowIndex = wordIndex > 0 ? rowIndex - wordIndex : rowIndex + verticalWord.length;
                // const finalRowIndex = wordIndex > 0 ? rowIndex - wordIndex : rowIndex;
                const placement = {
                    row: IndexIterator.getRowLetter(finalRowIndex),
                    col: this.board.translator.findColumnIndex(startNode.key),
                    direction: PlacementDirections.Vertical,
                    newWord: word,
                    oldWord: verticalWord,
                    letters: word.replace(verticalWord, ''),
                } as UserPlacement;
                if (finalRowIndex > 0 && finalRowIndex + placement.letters.length <= this.board.translator.rowCount)
                    this.registerPossiblePlacements(placement);
            }
        }
    }
 
    // Trouve les mots à placer dans la board
    private findSimilarWords(word: string) {
        let availableLetters = this.copyRack().getLetters();
        availableLetters += word;
        return this.findPossibleWords(word, availableLetters);
    }
 
    // availableLetters: les lettres du rack + lettres du mot existant
    private findPossibleWords(partial: string, availableLetters: string) {
        const possibleWords = [];
        const formableWords = [];
        // Again, don't use the instance of board. Create your own instance of DictionaryReader
        const dictionary = this.board.getDictionaryReader();
        // trouve les mots dans le dictionnaire qui contiennent le mot
        // trouve dans la board ===> Mots Formables
        for (const word of dictionary.getAllWords()) {
            if (word.startsWith(partial) || word.endsWith(partial)) formableWords.push(word);
        }
 
        // Trouve les mots à partir des mots formables, celle qu'on peut former avec les lettres du rack
        // ====> Mots Possibles
        for (const word of formableWords) {
            if (this.hasRequiredLetters(availableLetters, word)) possibleWords.push(word);
            // if (word.length > partial.length && this.hasRequiredLetters(availableLetters, word)) possibleWords.push(word);
        }
        return possibleWords;
    }
 
    private hasRequiredLetters(availableLetters: string, missingLetters: string): boolean {
        // Check if the player has the required letters to form the word
        const kMissingLetters = [...missingLetters];
        const kAvailableLetters = [...availableLetters];
 
        for (const letter of kMissingLetters) {
            if (kAvailableLetters.some((item) => item === letter)) {
                const idx = kAvailableLetters.findIndex((item) => item === letter);
                kAvailableLetters.splice(idx, 1);
            } else {
                return false;
            }
        }
        return true;
    }
 
    private registerPossiblePlacements(placement: UserPlacement) {
        if (placement.direction === PlacementDirections.Horizontal) {
            let incrementCol = 0;
            for (const letter of placement.newWord) {
                const colIndex = placement.col + incrementCol;
                this.board.placeLetter(letter, placement.row, colIndex, placement.direction);
                incrementCol++;
            }
            // let colIndex = placement.col;
            // for (const letter of placement.newWord) {
            //     this.board.placeLetter(letter, placement.row, colIndex, placement.direction);
            //     colIndex++;
            // }
        }
 
        if (placement.direction === PlacementDirections.Vertical) {
            let incrementRow = placement.row;
            for (const letter of placement.newWord) {
                this.board.placeLetter(letter, incrementRow, placement.col, placement.direction);
                incrementRow = IndexIterator.getNextRow(incrementRow) as string;
            }
        }
 
        if (this.board.askFormedWordsValidation(placement.row, placement.col, placement.direction)) {
            const points = this.board.getPoints(placement.row, placement.col, placement.direction);
            if (isNaN(points) === false && points > 0) {
                this.possiblePlacements.set(JSON.stringify(placement), points);
            }
        }
 
        this.board.restoreBoard();
    }

        return new Rack(this.rack.getLetters());

    private playBestAction(): string | undefined {
        console.log(this.possiblePlacements);
        if (this.possiblePlacements.size <= 0) {
            return `${FullCommandVerbs.SWITCH} ${this.rack.getLetters()}`;
        }
        if (this.level === GameLevel.Beginner) {
            const action = this.getAction();
            if (action === ActionIA.PassTurn) {
                return `${FullCommandVerbs.SKIP}`;
            }
            if (action === ActionIA.SwitchLetters) {
                return `${FullCommandVerbs.SWITCH} ${this.rack.getLetters()}`;
            }
            if (action === ActionIA.PlaceLetters) {
                const scoreInterval: ScoreInterval = this.getScoreInterval();
                let lastPlacement;
                for (const [key, score] of this.possiblePlacements) {
                    const placement = (lastPlacement = JSON.parse(key));
                    if (score >= scoreInterval.min && score <= scoreInterval.max) {
                        // const placement = JSON.parse(key);
                        console.log(placement);
                        return `${FullCommandVerbs.PLACE} ${placement.row}${placement.col}${placement.direction} ${placement.letters}`;
                    }
                }
 
                if (lastPlacement) {
                    const placement = lastPlacement;
                    return `${FullCommandVerbs.PLACE} ${placement.row}${placement.col}${placement.direction} ${placement.letters}`;
                }
            }
        }
 
        return `${FullCommandVerbs.SKIP}`;
    }
 
    private getAction(): string {
        if (this.actions.length <= 0) {
            this.actions = distribution<string>(
                [ActionIA.PlaceLetters, ActionIA.PassTurn, ActionIA.SwitchLetters],
                [WEIGHT_80, WEIGHT_10, WEIGHT_10],
                MAX_RANDOM,
            );
        }
 
        const index = Math.floor(this.actions.length * Math.random()); // random action
        const action: string = this.actions[index];
        this.actions.splice(index, 1); // on retire l<action qui vient detre executer
 
        return action;
    }
    private getScoreInterval(): ScoreInterval {
        if (this.scoreIntervals.length <= 0) {
            this.scoreIntervals = distribution<ScoreInterval>(
                [placeLetterIA.level0, placeLetterIA.level1, placeLetterIA.level2],
                [WEIGHT_40, WEIGHT_30, WEIGHT_30],
                MAX_RANDOM,
            );
        }
        const index = Math.floor(this.scoreIntervals.length * Math.random()); // random action
        const interval: ScoreInterval = this.scoreIntervals[index];
        this.scoreIntervals.splice(index, 1); // on retire l<interval qui vient detre executer
 
        return interval;
    }
